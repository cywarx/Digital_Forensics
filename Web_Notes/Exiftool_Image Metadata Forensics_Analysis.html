⌨ Lab Commands Reference
# Extract all metadata
$ exiftool -a -u -g1 forensic_scenario.jpg
# Show only GPS fields
$ exiftool -GPS:all forensic_scenario.jpg
# Check for hidden data
$ binwalk --extract forensic_scenario.jpg
# Dump strings from binary
$ strings forensic_scenario.jpg | grep -Ei "gps|lat|lon|phone|user|comment"
# Python: read GPS in code
$ python3 -c "import piexif; d=piexif.load('forensic_scenario.jpg'); print(d['GPS'])"
# Strip all metadata (privacy)
$ exiftool -all= -overwrite_original forensic_scenario.jpg
